\subsection*{\hypertarget{biotsavart}{Biot-Savartov zakon.}}
Zelo zanimiva stvar.

\begin{equation*}
    \Vec{B} = \frac{\mu_0I}{4\pi} \oint_S \frac{\di \Vec{s} \times \Vec{r}}{r^3}
\end{equation*}

V nevektorski obliki ga zapišemo kot
\begin{equation*}
    \di B = \frac{\mu_0I}{4\pi}  \frac{\di s \sin \alpha}{r^2},
\end{equation*}

pri čemer je $\alpha$  kot med tokovnim elementom $I \Vec{s}$ in radij vektorjem.

Z zakonom lahko izračunamo magnetno polje na oddaljenosti $a$ od neskončnega ravnega vodnika:
\begin{align*}
    B = \frac{\mu_0 I}{2\pi a}.
\end{align*}

Prav tako lahko izrazimo magnetno polje v središču krožne zanke s polmerom $r$:
\begin{align*}
    B = \frac{\mu_0 I}{2r}.
\end{align*}

In vodnika z dolžino $L$ na oddaljenosti $R$, pravokotno na krajišče vodnika
\begin{align*}
    B = \frac{\mu_0 I}{4\pi R} \frac{L}{\sqrt{L^2 + R^2}}.
\end{align*}

Funkcijo $B(z)$ za magnetno polje vzdolž simetrijske osi vodnika v obliki obroča s polmerom $R$. 
\begin{align*}
    B(z) = \frac{\mu_0 I}{2} \frac{R^2}{({R^2 + z^2})^{3/2}}.
\end{align*}
